149 template <class T>
150 bool areConsistent(QSharedPointer<SettingsBase> a, QSharedPointer<SettingsBase> b)
151 {
152 if(a->contains(m_key) && m_sb->setting<T>(m_key) == a->setting<T>(m_key))
153 a->remove(m_key);
154
155 if(b->contains(m_key) && m_sb->setting<T>(m_key) == b->setting<T>(m_key))
156 b->remove(m_key);
157
158 bool containsA = a->contains(m_key);
159 bool containsB = b->contains(m_key);
160
161 // returns true
162 // - if both settings bases do NOT contain the key
163 // - if both settings bases contain the SAME VALUE for the key
164 // returns false otherwise
165
166 return (containsA == containsB) && ((containsA && (a->setting<T>(m_key) == b->setting<T>(m_key))) || !containsA);
167 }
